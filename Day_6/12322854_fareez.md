# 27. Remove Element


**https://leetcode.com/problems/remove-element/?envType=problem-list-v2&envId=array**

```python
class Solution:
  def removeElement(self, nums: list[int], val: int) -> int:
    i = 0

    for num in nums:
      if num != val:
        nums[i] = num
        i += 1

    return i
```

# 66. Plus One


**https://leetcode.com/problems/plus-one/description/**

```python
class Solution:
  def plusOne(self, digits: list[int]) -> list[int]:
    for i, d in reversed(list(enumerate(digits))):
      if d < 9:
        digits[i] += 1
        return digits
      digits[i] = 0

    return [1] + digits
```

# 1. Two Sum

**https://leetcode.com/problems/two-sum/description/?q=two+sum**

```python
class Solution:
  def twoSum(self, nums: list[int], target: int) -> list[int]:
    numToIndex = {}

    for i, num in enumerate(nums):
      if target - num in numToIndex:
        return numToIndex[target - num], i
      numToIndex[num] = i
```

# Top K Frequent Elements

**https://neetcode.io/problems/top-k-elements-in-list/question?list=neetcode150**

```pyhton
from typing import List
from collections import Counter

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        count = Counter(nums)

        # buckets[i] contains numbers that appear i times
        buckets = [[] for _ in range(len(nums) + 1)]

        for num, freq in count.items():
            buckets[freq].append(num)

        res = []

        # Traverse from highest frequency to lowest
        for i in range(len(buckets) - 1, 0, -1):
            for num in buckets[i]:
                res.append(num)
                if len(res) == k:
                    return res
```
